# 1. Import necessary libraries
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import classification_report, confusion_matrix, accuracy_score

# Set style for plots
sns.set_style('whitegrid')
%matplotlib inline

# 2. Generate dummy economic data
np.random.seed(42)
n_samples = 1000

# Features: GDP growth (%), Inflation (%), Unemployment rate (%)
gdp_growth = np.random.normal(2.0, 1.5, n_samples)      # mean 2%, std 1.5%
inflation = np.random.normal(2.5, 1.0, n_samples)       # mean 2.5%, std 1%
unemployment = np.random.normal(5.0, 1.2, n_samples)    # mean 5%, std 1.2%

# Create target variable: economic state based on rules
# 0 = Recession (negative growth, high unemployment)
# 1 = Expansion (positive growth, low unemployment)
# 2 = Stagnation (low growth, moderate unemployment)

conditions = [
    (gdp_growth < 0) & (unemployment > 6.5),
    (gdp_growth > 2.5) & (unemployment < 5.0),
]
choices = [0, 1]  # recession and expansion
economic_state = np.select(conditions, choices, default=2)  # stagnation for all other cases

# Create DataFrame
df = pd.DataFrame({
    'GDP_growth': gdp_growth,
    'Inflation': inflation,
    'Unemployment': unemployment,
    'State': economic_state
})

# Map numeric states to labels for clarity
state_labels = {0: 'Recession', 1: 'Expansion', 2: 'Stagnation'}
df['State_label'] = df['State'].map(state_labels)

print(df.head())
print(df['State_label'].value_counts())

# 3. Visualize the data distribution
fig, axes = plt.subplots(1, 3, figsize=(15, 4))
sns.histplot(data=df, x='GDP_growth', hue='State_label', kde=True, ax=axes[0])
axes[0].set_title('GDP Growth Distribution by Economic State')
sns.histplot(data=df, x='Inflation', hue='State_label', kde=True, ax=axes[1])
axes[1].set_title('Inflation Distribution by Economic State')
sns.histplot(data=df, x='Unemployment', hue='State_label', kde=True, ax=axes[2])
axes[2].set_title('Unemployment Distribution by Economic State')
plt.tight_layout()
plt.show()

# 4. Prepare data for classification
X = df[['GDP_growth', 'Inflation', 'Unemployment']]
y = df['State']

# Train/test split
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42, stratify=y)

# 5. Train a Random Forest Classifier
clf = RandomForestClassifier(n_estimators=100, random_state=42)
clf.fit(X_train, y_train)

# 6. Predict and evaluate
y_pred = clf.predict(X_test)
accuracy = accuracy_score(y_test, y_pred)
print(f"\nAccuracy: {accuracy:.2f}")
print("\nClassification Report:")
print(classification_report(y_test, y_pred, target_names=['Recession', 'Expansion', 'Stagnation']))

# Confusion Matrix
cm = confusion_matrix(y_test, y_pred)
plt.figure(figsize=(6,5))
sns.heatmap(cm, annot=True, fmt='d', cmap='Blues', 
            xticklabels=['Recession', 'Expansion', 'Stagnation'],
            yticklabels=['Recession', 'Expansion', 'Stagnation'])
plt.title('Confusion Matrix')
plt.ylabel('Actual')
plt.xlabel('Predicted')
plt.show()

# 7. Feature Importance
importances = clf.feature_importances_
features = X.columns
plt.figure(figsize=(6,4))
sns.barplot(x=importances, y=features, palette='viridis')
plt.title('Feature Importance in Random Forest')
plt.xlabel('Importance')
plt.tight_layout()
plt.show()

# 8. Visualize predictions vs actual (scatter plot)
# Use first two features for 2D plot (GDP growth vs Unemployment)
plt.figure(figsize=(10,6))
scatter = plt.scatter(X_test['GDP_growth'], X_test['Unemployment'], 
                      c=y_pred, cmap='viridis', alpha=0.6, edgecolors='w', linewidth=0.5)
plt.xlabel('GDP Growth (%)')

def extract_images_from_website(url, output_file="nasa_images.txt"):
    """
    Extracts all image URLs from a given website and saves them to a file.

    Args:
        url (str): The website URL (e.g., "https://www.nasa.gov").
        output_file (str): The name of the file to save the URLs.
    """
    print(f"🌐 Fetching website: {url}")
    try:
        # Send a GET request to the website
        headers = {
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36'
        }
        response = requests.get(url, headers=headers, timeout=10)
        response.raise_for_status()  # Check for HTTP errors

        # Parse the HTML content
        soup = BeautifulSoup(response.text, 'html.parser')

        # Find all image tags
        img_tags = soup.find_all('img')
        print(f"🔍 Found {len(img_tags)} raw image tags.")

        if not img_tags:
            print("⚠️ No image tags found on the page.")
            return

        # Extract and process image sources
        image_urls = []
        base_url = f"{urlparse(url).scheme}://{urlparse(url).netloc}"

        for img in img_tags:
            # Get the 'src' attribute (or 'data-src' for lazy-loaded images)
            img_url = img.get('src') or img.get('data-src')
            if img_url:
                # Convert relative URLs to absolute URLs
                absolute_url = urljoin(base_url, img_url)
                # Filter out common unwanted data URLs or placeholders
                if absolute_url.startswith(('http', 'https')) and not absolute_url.startswith('data:'):
                    image_urls.append(absolute_url)

        # Remove duplicates by converting to a set and back to a list
        unique_urls = list(set(image_urls))
        print(f"✅ Extracted {len(unique_urls)} unique image URLs.")

        # Save the URLs to a file
        with open(output_file, 'w') as f:
            for img_url in unique_urls:
                f.write(img_url + '\n')

        print(f"📁 Image URLs saved to: {output_file}")
        return unique_urls

    except requests.exceptions.RequestException as e:
        print(f"❌ Error fetching the website: {e}")
    except Exception as e:
        print(f"❌ An unexpected error occurred: {e}")

# --- Helper function to remove emojis and non-Latin-1 characters ---
def sanitize_text(text):
    """Remove emojis and ensure text is Latin-1 compatible."""
    if not isinstance(text, str):
        text = str(text)
    # Remove emojis (Unicode blocks for emoticons, symbols, etc.)
    emoji_pattern = re.compile("["
        u"\U0001F600-\U0001F64F"  # emoticons
        u"\U0001F300-\U0001F5FF"  # symbols & pictographs
        u"\U0001F680-\U0001F6FF"  # transport & map symbols
        u"\U0001F700-\U0001F77F"  # alchemical symbols
        u"\U0001F780-\U0001F7FF"  # Geometric Shapes Extended
        u"\U0001F800-\U0001F8FF"  # Supplemental Arrows-C
        u"\U0001F900-\U0001F9FF"  # Supplemental Symbols and Pictographs
        u"\U0001FA00-\U0001FA6F"  # Chess Symbols
        u"\U0001FA70-\U0001FAFF"  # Symbols and Pictographs Extended-A
        u"\U00002702-\U000027B0"  # Dingbats
        u"\U000024C2-\U0001F251" 
        "]+", flags=re.UNICODE)
    text = emoji_pattern.sub(r'', text)
    # Replace any other problematic characters with '?'
    text = text.encode('latin-1', errors='replace').decode('latin-1')
    return text

class NASA_Missions_PDF(FPDF):
    def header(self):
        self.set_font('Arial', 'B', 16)
        self.cell(0, 10, sanitize_text('NASA Important Missions Report'), 0, 1, 'C')
        self.ln(10)
    
    def footer(self):
        self.set_y(-15)
        self.set_font('Arial', 'I', 8)
        self.cell(0, 10, sanitize_text(f'Page {self.page_no()} - Generated on {datetime.now().strftime("%Y-%m-%d")}'), 0, 0, 'C')
    
    def section_title(self, title):
        self.set_font('Arial', 'B', 12)
        self.set_fill_color(200, 220, 255)
        self.cell(0, 8, sanitize_text(title), 0, 1, 'L', 1)
        self.ln(4)
    
    def mission_entry(self, name, description, status="", year=""):
        self.set_font('Arial', 'B', 11)
        mission_header = name
        if year:
            mission_header += f" ({year})"
        if status:
            mission_header += f" - {status}"
        self.cell(0, 6, sanitize_text(mission_header), 0, 1, 'L')
        
        self.set_font('Arial', '', 10)
        self.multi_cell(0, 5, sanitize_text(description))
        self.ln(3)

def scrape_nasa_missions():
    """Scrape NASA missions from multiple authoritative sources"""
    
    missions_data = {
        "current_missions": [],
        "historical_missions": [],
        "future_missions": []
    }
    
    # Source 1: NASA Official Missions Page 
    print("🌐 Fetching current NASA missions...")
    try:
        url = "https://www.nasa.gov/nasa-missions/"
        headers = {'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'}
        response = requests.get(url, headers=headers, timeout=10)
        soup = BeautifulSoup(response.text, 'html.parser')
        
        featured = soup.find_all(['h2', 'h3'], string=re.compile('Featured Missions|Science|Human Spaceflight'))
        for section in featured:
            parent = section.find_parent()
            if parent:
                missions = parent.find_all(['h3', 'h4'], limit=6)
                for mission in missions:
                    name = mission.get_text().strip()
                    desc = mission.find_next('p')
                    if desc:
                        desc_text = desc.get_text().strip()[:300] + "..."
                        missions_data["current_missions"].append({
                            "name": name,
                            "description": desc_text,
                            "status": "Active"
                        })
    except Exception as e:
        print(f"⚠️ Error scraping NASA missions: {e}")
    
    # Source 2: Wikipedia comprehensive list 
    print("🌐 Fetching historical missions from Wikipedia...")
    try:
        wiki_url = "https://en.wikipedia.org/wiki/List_of_NASA_missions"
        response = requests.get(wiki_url, headers=headers, timeout=10)
        soup = BeautifulSoup(response.text, 'html.parser')
        
        tables = soup.find_all('table', class_='wikitable')
        if tables:
            rows = tables[0].find_all('tr')[1:8]  # First 7 programs
            for row in rows:
                cols = row.find_all('td')
                if len(cols) >= 5:
                    program = cols[0].get_text().strip()
                    start = cols[1].get_text().strip()
                    end = cols[3].get_text().strip()
                    desc = f"{program} program ran from {start} to {end}. {cols[4].get_text().strip()[:200]}"
                    missions_data["historical_missions"].append({
                        "name": program,
                        "description": desc,
                        "status": "Completed",
                        "year": f"{start}-{end}"
                    })
    except Exception as e:
        print(f"⚠️ Error scraping Wikipedia: {e}")
    
    # Source 3: Past Missions from NASA archive 
    print("🌐 Fetching past missions from NASA archive...")
    try:
        archive_url = "https://permanent.access.gpo.gov/lps71356/lps71356/www.nasa.gov/missions/past/index.html"
        response = requests.get(archive_url, headers=headers, timeout=10)
        soup = BeautifulSoup(response.text, 'html.parser')
        
        mission_links = soup.find_all('a', href=True)
        past_missions = ['Apollo', 'Gemini', 'Mercury', 'Skylab', 'Viking', 'Pioneer', 'Galileo']
        
        for mission_name in past_missions:
            for link in mission_links:
                if mission_name.lower() in link.get_text().lower():
                    desc = link.find_next('p')
                    if desc:
                        missions_data["historical_missions"].append({
                            "name": mission_name,
                            "description": desc.get_text().strip()[:300],
                            "status": "Completed"
                        })
                    break
    except Exception as e:
        print(f"⚠️ Error scraping archive: {e}")
    
    # Add key missions from National Academies report 
    critical_missions = [
        {
            "name": "Artemis Campaign",
            "description": "NASA's Artemis campaign aims to establish a sustainable human presence on the Moon, preparing for future Mars exploration. Includes SLS, Orion, and Lunar Gateway.",
            "status": "Ongoing",
            "year": "2017-present"
        },
        {
            "name": "Europa Clipper",
            "description": "Mission designed to study Jupiter's moon Europa, investigating whether the icy moon could have conditions suitable for life.",
            "status": "In Development",
            "year": "Launch ~2024"
        },
        {
            "name": "James Webb Space Telescope",
            "description": "Studies every phase in the history of our Universe, from first luminous glows after Big Bang to formation of solar systems.",
            "status": "Operational",
            "year": "2021-present"
        },
        {
            "name": "Mars Perseverance Rover",
            "description": "Collecting core samples of Martian rock and regolith for potential return to Earth, searching for signs of ancient life.",
            "status": "Operational",
            "year": "2020-present"
        }
    ]
    
    missions_data["current_missions"].extend(critical_missions)
    
    return missions_data

def generate_missions_pdf(missions_data, filename="NASA_Missions_Report.pdf"):
    """Generate a PDF report with all NASA missions"""
    
    pdf = NASA_Missions_PDF()
    pdf.add_page()
    
    # Introduction
    pdf.set_font('Arial', '', 11)
    intro = """This report contains information about NASA's most significant missions, 
both past and present, compiled from official NASA sources and public archives.
NASA's enduring purpose is scientific discovery and exploration for the benefit of the United States and all humanity."""
    pdf.multi_cell(0, 6, sanitize_text(intro))
    pdf.ln(5)
    
    # Current/Active Missions
    pdf.section_title("CURRENT & ACTIVE MISSIONS")  # emoji removed
    for mission in missions_data["current_missions"]:
        pdf.mission_entry(
            mission.get("name", "Unknown"),
            mission.get("description", "No description available."),
            mission.get("status", ""),
            mission.get("year", "")
        )
    
    # Historical Missions
    pdf.add_page()
    pdf.section_title("HISTORICAL MISSIONS (Completed)")  # emoji removed
    seen = set()
    unique_historical = []
    for mission in missions_data["historical_missions"]:
        if mission["name"] not in seen:
            seen.add(mission["name"])
            unique_historical.append(mission)
    
    for mission in unique_historical[:15]:
        pdf.mission_entry(
            mission.get("name", "Unknown"),
            mission.get("description", "Historical NASA mission."),
            mission.get("status", "Completed"),
            mission.get("year", "")
        )
    
    # Future/Planned Missions 
    pdf.add_page()
    pdf.section_title("FUTURE & PLANNED MISSIONS")  # emoji removed
    future = [
        {
            "name": "Artemis II",
            "description": "First crewed mission of the Artemis program, planned to launch four astronauts on a free-return flyby of the Moon in 2026.",
            "status": "Planned 2026"
        },
        {
            "name": "Artemis III",
            "description": "Will land the first astronauts on the lunar south pole, planned for 2026-2027.",
            "status": "Planned"
        },
        {
            "name": "NEO Surveyor",
            "description": "Near-Earth Object Surveyor mission to detect and characterize potentially hazardous asteroids.",
            "status": "Planned 2028"
        }
    ]
    
    for mission in future:
        pdf.mission_entry(mission["name"], mission["description"], mission["status"])
    
    # Summary statistics
    pdf.add_page()
    pdf.section_title("MISSION SUMMARY")  # emoji removed
    
    total_current = len(missions_data["current_missions"])
    total_historical = len(seen)
    total_future = len(future)
    
    summary = f"""Total missions covered: {total_current + total_historical + total_future}
• Current/Active Missions: {total_current}
• Historical Missions: {total_historical}
• Future/Planned Missions: {total_future}

NASA has successfully launched over 200 crewed flights since its establishment in 1958. 
The agency continues to push boundaries with programs like Artemis, aiming to establish 
sustainable lunar presence and prepare for human Mars exploration."""
    
    pdf.set_font('Arial', '', 11)
    pdf.multi_cell(0, 6, sanitize_text(summary))
    
    # Sources
    pdf.ln(10)
    pdf.set_font('Arial', 'B', 10)
    pdf.cell(0, 6, sanitize_text("Information Sources:"), 0, 1)
    pdf.set_font('Arial', '', 9)
    sources = [
        "• NASA Official Website - https://www.nasa.gov/nasa-missions/",
        "• Wikipedia List of NASA Missions",
        "• NASA Past Missions Archive",
        "• National Academies Press - NASA 20-Year Critical Missions"
    ]
    for source in sources:
        pdf.cell(0, 5, sanitize_text(source), 0, 1)
    
    # Save PDF
    pdf.output(filename)
    print(f"✅ PDF generated successfully: {filename}")
    return filename

def main():

if __name__ == "__main__":
